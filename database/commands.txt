# TestForge PostgreSQL

cd /home/bruce/Projects/testforge-v1.2/database

# Start Database
docker-compose up -d

# Check Status
docker ps

# View Logs
docker logs -f testforge-postgres

# Connect to Database
docker exec -it testforge-postgres psql -U testforge -d testforge -c '\dt'

# Stop Database
docker-compose down

# Reset Database (Deletes All Data)
docker exec -it testforge-postgres psql -U testforge -d testforge -c "
